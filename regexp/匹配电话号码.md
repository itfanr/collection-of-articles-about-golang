# go语言匹配电话号码

标签（空格分隔）： regexp

---

```
package main
import (
    "fmt"
    "regexp"
    )

const PHONE_RGX = `^1([38][0-9]|14[57]|5[^4])\d{8}$`

func main() {
	rgx := regexp.MustCompile(PHONE_RGX)
	s := []string{"18505921256", "13849594009", "18759029321"}
	for _, num := range s {
		fmt.Println(rgx.MatchString(num))
	}
}

```
